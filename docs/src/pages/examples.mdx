# Examples

Some examples and use cases

## Creating your own `<EChart />` component

```tsx copy
import { useECharts, type UseEChartsOptions } from '@kbox-labs/react-echarts'

export const EChart: FC<UseEChartsOptions> = props => {
  const [ref, echartsInstance] = useECharts<HTMLDivElement>(props)

  return <div ref={ref} />
}
```
